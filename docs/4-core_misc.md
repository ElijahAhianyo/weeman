def printt(s, msg):
    if s == 1:
        print('\x1b[01;31m[%s] Error: %s\x1b[00m' % (time.strftime('%H:%M:%S'), msg))
        sys.exit(1)
    elif s == 2:
        print('\x1b[01;32m[%s] %s\x1b[00m' % (time.strftime('%H:%M:%S'), msg))
    elif s == 3:
        print('\x1b[01;01m[%s] %s\x1b[00m' % (time.strftime('%H:%M:%S'), msg))
    else:
        print('\x1b[01;01m[%s] %s\x1b[00m' % (time.strftime('%H:%M:%S'), msg))
