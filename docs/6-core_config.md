```python
def history_getkey(key):
    try:
        history = open('history.log', 'r').readlines()
    except Exception as e:
        return 0
    if history == None:
        return 0
    for line in history:
        if line.startswith('\n') or line.startswith('#'):
            pass
        (skey, value) = line.split(' = ')
        if skey == key:
            return str(value[:-1])
    return 0
```
